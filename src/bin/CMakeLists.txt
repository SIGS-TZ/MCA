function(mca_add_executable name srcs)
    add_executable(${name} ${srcs})
    target_compile_features(${name} PRIVATE cxx_std_20)
    target_link_libraries(${name} PRIVATE mca)
    copy_dlls_if_needed(${name})
endfunction()

mca_add_executable(dbg_draw_circles "dbg_draw_circles.cpp")
mca_add_executable(dbg_draw_used_area "dbg_draw_used_area.cpp")
mca_add_executable(dbg_backward "dbg_backward.cpp")
mca_add_executable(dbg_forward "dbg_forward.cpp")
mca_add_executable(main_preproc "main_preproc.cpp")
mca_add_executable(main_postproc "main_postproc.cpp")
