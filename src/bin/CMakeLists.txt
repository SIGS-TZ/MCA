function(mca_add_executable name srcs)
    add_executable(${name} ${srcs})
    target_link_libraries(${name} PRIVATE libMCA)
    if (BUILD_SHARED_LIBS)
        add_custom_command(TARGET ${name} POST_BUILD COMMAND ${CMAKE_COMMAND} -E copy_if_different $<TARGET_FILE:libMCA> $<TARGET_FILE_DIR:${name}>)
    endif ()
endfunction()
